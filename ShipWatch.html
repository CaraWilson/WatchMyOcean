async function updateShipView() { const datasetID = document.getElementById('ship-selector').value; const stats = document.getElementById('ship-stats'); const satImg = document.getElementById('sat-image'); const loading = document.getElementById('loading-msg'); if (!datasetID) return; loading.style.display = 'block'; loading.innerText = "Querying ERDDAP..."; satImg.style.display = 'none'; // We use .jsonp (JSON with Padding) if .json fails, // but first let's try the standard JSON with a cache-buster const erddapUrl = `https://coastwatch.pfeg.noaa.gov/erddap/tabledap/${datasetID}.json?time%2Clatitude%2Clongitude&orderByMax(%22time%22)`; try { const response = await fetch(erddapUrl); if (!response.ok) { throw new Error(`ERDDAP responded with status: ${response.status}`); } const data = await response.json(); // Extracting data from ERDDAP's nested JSON structure const columnNames = data.table.columnNames; const rows = data.table.rows; if (rows.length === 0) { throw new Error("No data found for this ship."); } // Mapping columns to indices const timeIdx = columnNames.indexOf('time'); const latIdx = columnNames.indexOf('latitude'); const lonIdx = columnNames.indexOf('longitude'); const lat = rows[0][latIdx]; const lon = rows[0][lonIdx]; const time = rows[0][timeIdx]; stats.innerHTML = `Ship: ${datasetID}
Lat: ${lat.toFixed(4)} | Lon: ${lon.toFixed(4)}`; // Define Bounding Box for the Satellite Image const delta = 0.5; // size of the box in degrees const minLat = lat - delta; const maxLat = lat + delta; const minLon = lon - delta; const maxLon = lon + delta; // Using the NRT (Near Real Time) VIIRS Chlorophyll layer as an example const satUrl = `https://coastwatch.pfeg.noaa.gov/erddap/wms/pfehVIIRSnrt/request?SERVICE=WMS&VERSION=1.3.0&REQUEST=GetMap&LAYERS=pfehVIIRSnrt:chlor_a&STYLES=etc&CRS=EPSG:4326&BBOX=${minLat},${minLon},${maxLat},${maxLon}&WIDTH=800&HEIGHT=600&FORMAT=image/png`; satImg.src = satUrl; satImg.onload = () => { loading.style.display = 'none'; satImg.style.display = 'block'; }; } catch (error) { console.error("Full Error Details:", error); loading.innerText = "Error: " + error.message; } }
