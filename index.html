from erddapy import ERDDAP import matplotlib.pyplot as plt import cartopy.crs as ccrs import cartopy.feature as cfeature import pandas as pd # 1. Setup ERDDAP connection e = ERDDAP(server="https://coastwatch.pfeg.noaa.gov/erddap", protocol="tabledap") # 2. Fetch Station Data e.dataset_id = "erdCalCOFIstns" e.constraints = { "cruise=": "202304", "line>": 84, "station<=": 70, } e.variables = ["longitude", "latitude", "station", "line"] df_stns = e.to_pandas().dropna() # 3. Fetch Satellite Data (Example: Daily Chl-a) # We'll use a simple URL request for the grid to keep it lightweight grid_url = "https://coastwatch.pfeg.noaa.gov/erddap/griddap/nesdisVHNnoaa20chlaDaily.nc?chlor_a[(last)][(0.0)][(29):(39)][(-127):(-116)]" # (In a full script, you'd use xarray to open this .nc file) # 4. Plotting fig = plt.figure(figsize=(10, 8)) ax = plt.axes(projection=ccrs.PlateCarree()) ax.set_extent([-127, -116, 29, 39]) # Add map features ax.add_feature(cfeature.COASTLINE) ax.add_feature(cfeature.LAND, facecolor='lightgray') # Plot Stations plt.scatter(df_stns['longitude (degrees_east)'], df_stns['latitude (degrees_north)'], color='yellow', edgecolors='black', s=50, transform=ccrs.PlateCarree(), label='CalCOFI Stations') plt.title("CalCOFI Cruise 202304 - Station Map") plt.legend() plt.show()
