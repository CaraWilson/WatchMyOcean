// Plot-box tuning specifically for C-HARMS panels function getPlotBox(canvas, id) { const W = canvas.width, H = canvas.height; // Default margins for standard products let L = 0.15, R = 0.05, T = 0.13, B = 0.38; // C-HARMS Specific Calibration: // These maps have wider left margins (L) to accommodate the Lat labels // and different vertical offsets (T/B) if (id.includes("charms")) { L = 0.088; // Shifted left to match C-HARMS Y-axis R = 0.045; // Right margin T = 0.115; // Top margin (Header) B = 0.225; // Bottom margin (Colorbar) } else if (["chla-d", "chla-w", "sst", "sardine", "ssh", "wind", "sss"].includes(id)) { L = 0.075; B = 0.225; } return { left: Math.round(W * L), top: Math.round(H * T), width: Math.max(10, Math.round(W * (1 - L - R))), height: Math.max(10, Math.round(H * (1 - T - B))) }; } function drawStations(canvasId, imgId, sLat, nLat, wLon, eLon) { const canvas = document.getElementById(canvasId), img = document.getElementById(imgId); if (!canvas || !img || img.clientWidth === 0) return; canvas.width = img.clientWidth; canvas.height = img.clientHeight; const ctx = canvas.getContext("2d"); ctx.clearRect(0, 0, canvas.width, canvas.height); const isDefault = (sLat === 29 && nLat === 39 && wLon === -127 && eLon === -116); if (!isDefault) return; const id = canvasId.replace("cvs-", ""); const box = getPlotBox(canvas, id); // CRITICAL: C-HARMS is clamped to 31.3N - 43N and -127.5W - -117W. // We must use these bounds for the math, not the user input 29-39. let curSLat = sLat, curNLat = nLat, curWLon = wLon, curELon = eLon; if (id.includes("charms")) { curSLat = 31.3; curNLat = 39.01; // Matches the default view top-crop for C-HARMS curWLon = -127.0; curELon = -117.0; } ctx.save(); ctx.beginPath(); ctx.rect(box.left, box.top, box.width, box.height); ctx.clip(); STATIONS.forEach(p => { // Mapping station Lat/Lon to the specific Grid Bounds of the C-HARMS panel const px = { x: box.left + ((p.lon - curWLon) / (curELon - curWLon)) * box.width, y: box.top + ((curNLat - p.lat) / (curNLat - curSLat)) * box.height }; ctx.beginPath(); ctx.arc(px.x, px.y, 3.5, 0, Math.PI * 2); ctx.fillStyle = "yellow"; ctx.fill(); ctx.lineWidth = 1; ctx.strokeStyle = "black"; ctx.stroke(); }); ctx.restore(); }
